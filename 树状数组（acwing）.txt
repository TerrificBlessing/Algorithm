#include <iostream>
#include <cstdio>
#include <cstring>

const int N = 100000 + 10;

int tr[N];
int n, m;

int lowbit(int x)
{
    return x & -x;
}

void add(int x, int v)
{
    for(int i = x; i <= n; i += lowbit(i)) tr[i] += v; 
}

int query(int x)
{
    int res = 0;
    for(int i = x; i; i -= lowbit(i)) res += tr[i];

    return res;
}

int main()
{

    scanf("%d %d", &n, &m);

    for(int i = 1; i <= n; ++ i)
    {
        int x;
        scanf("%d", &x);
        add(i, x);
    }

    for(int i = 1; i <= m; ++ i)
    {
        int k, a, b;
        scanf("%d %d %d", &k, &a, &b);
        if(k == 0) printf("%d\n", query(b) - query(a - 1));
        else add(a, b);
    }

    return 0;
}
