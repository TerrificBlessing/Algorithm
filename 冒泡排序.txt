#include<iostream>
using namespace std;
int main()
{
	void swap(int&, int&);
	int a[20],*p=a;
	int n;
	cin >> n;
	int i, j;
	for (i = 0; i <= n - 1; i++)
		cin >> *(p + i);
	for (j = 1; j <= n - 1;j++)
	{
		for (i = 1; i <= n - 1; i++)
			if (*(p + i - 1) > *(p + i)) swap(*(p + i - 1), *(p + i));
	}
	for (i = 0; i <= n - 1; i++)
		printf("%d ", *(p + i));
	return 0;
}
void swap(int &x, int  &y)
{
	int temp;
	temp = x;
	x = y;
	y = temp;
}
