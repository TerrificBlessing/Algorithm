#include <iostream>
#include <cstdio>
#include <cstring>
#include <algorithm>

using namespace std;

struct Node
{
	int u, v, w;
	Node()
	{
		u = v = w = -1;
	}
	Node(int a, int b, int c): u(a), v(b), w(c) {}

}st[202000];

int dist[1010];

int main()
{
	int N, ML, MD;
	cin >> N >> ML >> MD;
	int cnt = 0;
	int inf = 2000000000;
	for (int i = 1; i <= N; ++i)
	{
		dist[i] = inf;
	}
	dist[1] = 0;
	for (int i = 2; i <= N; ++i)
	{
		st[cnt++] = Node(i, i - 1, 0);
	}
	for (int i = 1; i <= ML; ++i)
	{
		int A, B, D;
		scanf("%d%d%d", &A, &B, &D);
		st[cnt++] = Node(A, B, D);
	}
	for (int i = 1; i <= MD; ++i)
	{
		int A, B, D;
		scanf("%d%d%d", &A, &B, &D);
		st[cnt++] = Node(B, A, -D);
	}
	bool flag;
	for (int i = 1; i <= N - 1; ++i)
	{
		flag = false;
		for (int j = 0; j < cnt; ++j)
		{
			int a = st[j].u, b = st[j].v, c = st[j].w;
			if (dist[b] > dist[a] + c)
			{
				dist[b] = dist[a] + c;
				flag = true;
			}
		}
		if (!flag) break;
	}
	int res = 0;
	for (int i = 0; i < cnt; ++i)
	{
		int a = st[i].u, b = st[i].v, c = st[i].w;
		if (dist[b] > dist[a] + c)
		{
			res = -1;
			break;
		}
	}
	if (res == 0)
	{
		if (dist[N] >= 2e9) res = -2;
	}
	if (res == 0)
	{
		res = dist[N];
	}
	cout << res << endl;
	return 0;
}
