#include <iostream>
#include <cstdio>
#include <cstring>
#include <algorithm>
#include <vector>
#include <queue>

using namespace std;

struct Node
{
	int to, cost;
	Node(int a, int b) :to(a), cost(b) {}
	friend bool operator < (const Node& n1, const Node& n2)
	{
		return n1.cost < n2.cost;
	}
};
bool used[20010];

int main()
{
	int TT;
	scanf("%d", &TT);
	while (TT--)
	{
		int N, M, R;
		scanf("%d%d%d", &N, &M, &R);
		vector<Node> G[20010];

		memset(used, false, sizeof used);
		priority_queue<Node> que;

		for (int i = 0; i < R; ++i)
		{
			int a, b, c;
			scanf("%d%d%d", &a, &b, &c);
			G[a].push_back(Node(b + N, c));
			G[b + N].push_back(Node(a, c));
		}
		int res = 0;
		for (int i = 0; i < N + M; ++i)
		{
			que.push(Node(i, 0));
		}
		while (que.size())
		{
			Node n = que.top();
			que.pop();
			if (!used[n.to]) res += n.cost;
			used[n.to] = true;
			vector<Node>::iterator it = G[n.to].begin();
			for (; it != G[n.to].end(); ++it)
			{
				if (!used[it->to]) que.push(*it);
			}

		}
		int ans = (N + M) * 10000 - res;
		printf("%d\n", ans);

	}

	return 0;
}
