#include <iostream>
#include <cstring>
#include <cstdio>
#include <algorithm>

using namespace std;

typedef long long LL;



LL exgcd(LL a, LL b, LL &x, LL &y)
{
    if (!b) 
    {
        x = 1, y = 0;
        return a;
    }

    LL d = exgcd(b, a % b, x, y);
    LL temp = x;
    x = y;
    y = temp - a / b * y;


    return d;
}


int main()
{
    int T;
    cin >> T;
    while(T -- )
    {
        LL n, d, x, y;
        cin >> n >> d >> x >> y;
        LL k1, k2;
        LL v = exgcd(d, n, k1, k2);
        k2 = - k2;

        if((y - x) % v != 0) 
        {
            cout << "Impossible" << endl;
            continue;
        }
        k1 *= (y - x) / v;
        LL t = n / v;

        cout << (k1 % t + t) % t << endl;

    }


    return 0;

}
