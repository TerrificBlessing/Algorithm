#include <iostream>
#include <cstdio>
#include <cstring>
#include <cmath>
#include <algorithm>
#include <vector>
#pragma warning(disable:4996)

using namespace std;

const int N = 500000 + 10;
typedef pair<int, int> PII;


vector<int> e[N];
vector<PII> query[N];
int fa[N];
bool vis[N];
int ans[N];


int find(int u)
{
	if (u == fa[u]) return u;
	
	fa[u] = find(fa[u]);

	return fa[u];
}

void tarjan(int u)
{
	vis[u] = true;
	for (auto v : e[u])
	{
		if (!vis[v])
		{
			tarjan(v);
			fa[v] = u;
		}
	}

	for (auto q : query[u])
	{
		int v = q.first;
		int i = q.second;
		if (vis[v])
		{
			ans[i] = find(v);
		}
	}

}

int main()
{
	int n, m, root;
	scanf("%d %d %d", &n, &m, &root);
	
	for (int i = 1; i <= n - 1; ++i)
	{
		int a, b;
		scanf("%d %d", &a, &b);
		e[a].push_back(b);
		e[b].push_back(a);
	}

	for (int i = 1; i <= m; ++i)
	{
		int a, b;
		scanf("%d %d", &a, &b);
		query[a].push_back({b, i});
		query[b].push_back({a, i});
	}

	for (int i = 1; i <= n; ++i)
	{
		fa[i] = i;
	}

	tarjan(root);

	for (int i = 1; i <= m; ++i)
	{
		printf("%d\n", ans[i]);


	}

	return 0;

}
