#include <iostream>
#include <cstdio>
#include <algorithm>
#include <cmath>

using namespace std;

#define LL long long

const int N = 31;
const int M = 100;

int n, C;
int w[N];
int p[N];
bool st[N];
int res;
bool flag = false;
int dp[N][M];

void dfs(int x, int sum, int weight)
{
	if (flag) return;

	if (x == 0 && sum == res)
	{
		flag = true;
		for (int i = 1; i <= n; ++i)
		{
			if (st[i]) cout << 1;
			else cout << 0;
		}
		cout << endl;

		return;
	}

	if (x == 0)
	{
		return;
	}

	dfs(x - 1, sum, weight);
	st[x] = true;
	dfs(x - 1, sum + p[x], weight + w[x]);
	st[x] = false;

}

int main()
{
	cin >> n;
	cin >> C;

	for (int i = 1; i <= n; ++i)
	{
		cin >> w[i];
	}

	for (int i = 1; i <= n; ++i)
	{
		cin >> p[i];
	}
	
	for (int i = 1; i <= n; ++i)
	{
		for (int j = 1; j <= C; ++j)
		{
			if (w[i] <= j)
			{
				dp[i][j] = max(dp[i - 1][j - w[i]] + p[i], dp[i - 1][j]);
			}
			else
				dp[i][j] = dp[i - 1][j];

		}
	}
	
	res = dp[n][C];
	cout << res << " ";

	dfs(n, 0, 0);

	

	return 0;

}
