#include <iostream>
#include <cstdio>
using namespace std;
const int N = 100000 + 10;
int arr[N];
int dif[N];
int main()
{
    int n, m;
    scanf("%d%d", &n, &m);

    for(int i = 1; i <= n; ++ i)
    {
        scanf("%d", &arr[i]);
        dif[i] = arr[i] - arr[i - 1];
    }
    for(int i = 1; i <= m; ++ i)
    {
        int l, r, c;
        scanf("%d%d%d", &l, &r, &c);
        dif[l] += c;
        dif[r + 1] -= c;
    }
    int x = 0;
    for(int i = 1; i <= n; ++ i)
    {
        x = dif[i] + x;
        printf("%d ", x);
    }



    return 0;

}
