#include<bits/stdc++.h> 

using namespace std;
typedef long long LL;
const int N = 100000 + 10;

int st[3][N];
int n;

LL calx()
{
    LL ans = 0;
    int midy = st[1][n / 2];
    int midz = st[2][n / 2];
    for (int i = 0; i < n; ++ i)
    {
        ans += abs(st[1][i] - midy);
        ans += abs(st[2][i] - midz);
    }
    int midx = st[0][n / 2] - n / 2 - 1;
    for (int i = 0; i < n; ++ i)
    {
        ans += abs(st[0][i] - i - 1 - midx);
    }

    return ans;
}

LL caly()
{
    LL ans = 0;
    int midx = st[0][n / 2];
    int midz = st[2][n / 2];
    for (int i = 0; i < n; ++ i)
    {
        ans += abs(st[0][i] - midx);
        ans += abs(st[2][i] - midz);
    }
    int midy = st[1][n / 2] - n / 2 - 1;
    for (int i = 0; i < n; ++ i)
    {
        ans += abs(st[1][i] - i - 1 - midy);
    }

    return ans;
}
LL calz()
{
    LL ans = 0;
    int midx = st[0][n / 2];
    int midy = st[1][n / 2];
    for (int i = 0; i < n; ++ i)
    {
        ans += abs(st[0][i] - midx);
        ans += abs(st[1][i] - midy);
    }
    int midz = st[2][n / 2] - n / 2 - 1;
    for (int i = 0; i < n; ++ i)
    {
        ans += abs(st[2][i] - i - 1 - midz);
    }

    return ans;
}

int main( )
{
   
    cin >> n;
    
    for (int i = 0; i < n; ++ i)
    {
        cin >> st[0][i] >> st[1][i] >> st[2][i];
    }
    for (int i = 0; i < 3; ++ i)
    {
        sort(st[i], st[i] + n);
    }
    LL res_x = calx();
    LL res_y = caly();
    LL res_z = calz();
    cout << min(res_x, min(res_y, res_z)) << endl;
    return 0;
}
