#include<iostream>
using namespace std;
int main()
{
	void swap(int&, int&);
	int n;
	cin >> n;
	int a[100];
	int i, * p = a;
	for (i = 0; i <= n - 1; i++)
		cin >> *(p + i);
	int j, *k;
	for ( j = 1; j <= n -1; j++)
	{
		k = p;
		for (i = 1; i <= n -j; i++)
			if (*(p + i) < *k) k = p + i;
		swap(*p, *k);
		p++;
	}
	int* g = a;
	for (i = 0; i <= n - 1; i++)
		printf("%d ", *(g + i));
	return 0;
}
void swap(int& x, int& y)
{
	int temp;
	temp = x;
	x = y;
	y = temp;
}
