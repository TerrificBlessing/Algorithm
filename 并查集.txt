#include <iostream>
#include <cstdio>
#include <cstring>

using namespace std;
const int N = 10;

int fa[N];
int n;

void init()
{
	for (int i = 0; i < n; ++i)
	{
		fa[i] = i;
	}
}

int find(int x) //路径压缩
{
	if (fa[x] == x) return x;

	fa[x] = find(fa[x]);

	return fa[x];
}


void Union(int x, int y)
{
	int x_fa = find(x);
	int y_fa = find(y);

	fa[y_fa] = x_fa;

}

bool Is_Circle(int a, int b)
{
	if (find(a) == find(b)) return true;
	
	return false;

}

int main()
{
	cin >> n;

	init();

	cout << Is_Circle(1, 3) << endl;
	Union(1, 2);
	cout << Is_Circle(1, 3) << endl;
	Union(2, 3);
	cout << Is_Circle(1, 3) << endl;

	return 0;

}
