#include <iostream>
#include <cstdio>
#include <cstring>
#include <algorithm>
#include <queue>

using namespace std;

//A:0,B:1,C:2...
const int N = 30;
int fa[N]; //并查集

struct EdgeNode
{
	char u, v;
	int weight;
	friend bool operator < (const EdgeNode& e1, const EdgeNode& e2)
	{
		return e1.weight > e2.weight;
	}
};

void init()
{
	for (int i = 0; i < 30; ++i)
		fa[i] = i;

}

int find(int x)
{
	if (fa[x] == x) return x;
	fa[x] = find(fa[x]);//路径压缩
	return fa[x];
}

void Union(EdgeNode& e)
{
	char u1 = e.u, u2 = e.v;
	int fa_u1 = find(u1 - 'A');
	int fa_u2 = find(u2 - 'A');
	fa[fa_u2] = fa_u1;
}

bool IsCircle(EdgeNode& e)
{
	return find(e.u - 'A') == find(e.v - 'A');
}
int Kruskal(int n, priority_queue<EdgeNode> &que)
{
	int sum = 0;
	int count = 0;//记录加入的边数
	init();//每次初始化并查集
	while (count != n - 1 && !que.empty())
	{
		EdgeNode e = que.top();
		que.pop();
		if (IsCircle(e)) continue;
		Union(e);
		count++;
		sum += e.weight;
	}
	
	if (count == n - 1) return sum;
	else return -1;//没有MST
}

int main()
{
	int n = -1;
	while (1)
	{
		cin >> n;
		if (n == 0) break;
		priority_queue<EdgeNode> que;//每次重新创建一个优先队列
		for (int i = 0; i < n - 1; ++i)
		{
			char u;
			cin >> u;//按顺序出现的字母
			int m;
			cin >> m;
			for (int i = 1; i <= m; ++i)
			{
				char v;
				cin >> v;
				int cost;
				cin >> cost;
				que.push({ u, v, cost });
			}
		}
		cout << Kruskal(n, que) << endl;
	}

	return 0;
}
