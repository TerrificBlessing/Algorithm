#include <iostream>
#include <cstdio>
using namespace std;

const int N = 1001, M = 1001, Q = 200001;

int arr[N][M];
int ans[Q];


int main()
{
    int n, m, q;
    
    cin >> n >> m >> q;
    
    for(int i = 0; i <= n; ++ i)
    {
        arr[0][i] = 0;
        arr[i][0] = 0;
    }
    
    for(int i = 1; i <= n; ++ i)
        for(int j = 1; j <= m; ++j)
            {
                int x;
                cin >> x;
                arr[i][j] = arr[i - 1][j] + arr[i][j - 1] - arr[i - 1][j - 1] + x;
            }
    
    for(int i = 0; i < q; ++ i)
    {
        int x1, y1, x2, y2;
        cin >> x1 >> y1 >> x2 >> y2;
        ans[i] = arr[x2][y2] - arr[x1 - 1][y2] - arr[x2][y1 - 1] + arr[x1 - 1][y1 - 1];
        
        
    }
    
    for(int i = 0; i < q; ++ i)
    {
        cout << ans[i] << endl;
    }
    
    
    return 0;
            
    
}
