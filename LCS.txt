#include<iostream>
#include<algorithm>
using namespace std;
int a[100000], b[100000] ,m[100000],q[100000],dp[100000];
const int INF = 100000000;

int main()
{
	int n, k;
	cin >> n;
	for (int i = 1; i <= n; i++)
	{
		cin >> a[i]; 
		m[a[i]] = i;
	}
	for (int i = 1; i <= n; i++)
	{
		cin >> b[i];
		q[b[i]] = i;
	}
	for (int i = 1; i <= n; i++)
		b[q[i]] = m[i];
	fill( dp+1,dp + n+1,INF );
	for (int i = 1; i <= n; i++)
		*lower_bound(dp + 1, dp + n + 1, b[i]) = b[i];
	printf("%d", lower_bound(dp + 1, dp + n+1, INF) - (dp+1));
	return 0;

	
	
	
}
