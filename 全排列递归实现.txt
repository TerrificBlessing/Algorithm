#include <iostream>
#include <cstdio>
#include <algorithm>

using namespace std;
const int N = 10;

int st[N];
bool used[N];
int n;

void dfs(int x)
{
    if (x >= n)
    {
        for (int i = 0; i < n; ++ i)
        {
            cout << st[i];
        }
        cout << endl;
      	return;  
    }
    
    
    for(int i = 1; i <= n; ++ i)
    {
        if(!used[i])
        {
            st[x] = i;
            used[i] = 1;
            dfs(x + 1);
            st[x] = 0;
            used[i] = 0;
        }
    }
    
}


int main()
{
 
    cin >> n;
    dfs(0);
    return 0;
    
    
}
