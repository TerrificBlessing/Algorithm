#include <iostream>
#include <cstdio>


using namespace std;

const int N = 100010;
int tem[N];

void merge_sort(int q[], int l, int r)
{
    if(l >= r) return;
    
    int mid = l + r >> 1;
    merge_sort(q, l, mid);
    merge_sort(q, mid + 1, r);
    
    int k = 0;
    int i = l;
    int j = mid + 1;
    while(i <= mid && j <= r)
    {
        if(q[i] < q[j]) tem[k ++] = q[i ++];
        else tem[k ++] = q[j ++];
    }
    
    while(i <= mid) tem[k ++] = q[i ++];
    while(j <= r) tem[k ++] = q[j ++];
    
    for(i = l, j = 0; i <= r; ++ i, ++ j) q[i] = tem[j];
    
}


int main()
{
    int arr[N];
    
    int n;
    
    scanf("%d", &n);
    
    for(int i = 0; i < n; ++ i)
    {
        scanf("%d", arr + i);
    }
    
    merge_sort(arr, 0, n - 1);
    
    for(int i = 0; i < n; ++ i)
    {
        printf("%d ", arr[i]);
    }
    
    return 0;
}
