#include <iostream>
#include <cstdio>
#include <cstring>
#include <limits.h>

using namespace std;
const int N = 4 * 100000;

int arr[N];
int n, m;

void Tree_Insert(int x, int pos, int l, int r, int i)
{
    arr[pos] = max(arr[pos], x);

    if(i == l && i == r) return;

    int mid = l + r >> 1;

    if(i <= mid) Tree_Insert(x, 2 * pos + 1, l, mid, i);
    else Tree_Insert(x, 2 * pos + 2, mid + 1, r, i);



}

int Tree_Search(int a, int b, int l, int r, int pos)
{
   if(l >= a && r <= b) return arr[pos];
   if(r < a || l > b) return INT_MIN;

   int mid = l + r >> 1;

   return max(Tree_Search(a, b, l, mid, pos * 2 + 1), Tree_Search(a, b, mid + 1, r, pos * 2 + 2));


}

int main()
{

    scanf("%d %d", &n, &m);

    for(int i = 0; i < N; ++ i)
    {
        arr[i] = INT_MIN;
    }


    for(int i = 1; i <= n; ++ i)
    {
        int x;
        scanf("%d", &x);
        Tree_Insert(x, 0, 1, n, i);
    }

    for(int i = 1; i <= m; ++ i)
    {
        int a, b;
        scanf("%d %d", &a, &b);
        int res = Tree_Search(a, b, 1, n, 0);
        printf("%d\n", res);
    }


    return 0;
}
