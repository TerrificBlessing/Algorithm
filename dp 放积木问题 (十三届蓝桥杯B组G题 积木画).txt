//内存限制只能过9/10，要AC的话要用动态dp
#include <iostream>
#include <cstdio>
#include <cstring>

using namespace std;
const int N = 10000000 + 10;
const int MOD = 1000000007;
typedef long long LL;

LL f[N][4];

int main()
{
    int n;
    cin >> n;
    f[0][0] = 1;
    for (int i = 1; i <= n; ++ i)
    {
        if (i == 1)
        {
            f[i][0] = f[i - 1][0];
            f[i][1] = f[i - 1][0] + f[i - 1][2] + f[i - 1][3];
            f[i][2] = f[i - 1][0] + f[i - 1][3];
            f[i][3] = f[i - 1][0] + f[i - 1][2];
        }
        else
        {
            f[i][0] = (f[i - 1][0] + f[i - 2][1]) % MOD;
            f[i][1] = (f[i - 1][0] + f[i - 2][1] + f[i - 1][2] + f[i - 1][3]) % MOD;
            f[i][2] = (f[i - 1][0] + f[i - 2][1] + f[i - 1][3]) % MOD;
            f[i][3] = (f[i - 1][0] + f[i - 2][1] + f[i - 1][2]) % MOD;
        }
        
    }

    cout << (f[n - 1][1] + f[n][0]) % MOD << endl;
    
    return 0;
}
